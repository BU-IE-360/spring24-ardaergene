from evds import evdsAPI
evds = evdsAPI('E2ZSqPaQz5')
x = evds.get_data(['TP.DK.USD.A.YTL','TP.AKONUTSAT1.TOPLAM', 'TP.YISGUCU2.G2'], startdate="01-01-2019", enddate="01-01-2022")
df = pd.DataFrame(x)
print(df.head())
plt.figure(figsize=(15, 5))
plt.plot(df['Tarih'], df['TP_DK_USD_A_YTL'], label='TP_DK_USD_A_YTL')
plt.xlabel('Tarih')
plt.ylabel('TP_DK_USD_A_YTL')
plt.title('TP_DK_USD_A_YTL')
plt.xticks(rotation=45)
plt.legend()
plt.show()
result = seasonal_decompose(df['TP_DK_USD_A_YTL'], model='additive', period=12)
result.plot()
plt.show()
corr_matrix = df.corr()
sns.heatmap(corr_matrix, annot=True, cmap='coolwarm')
plt.title('Correlation Matrix')
plt.show()
